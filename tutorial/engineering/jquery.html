与jQuery的混用

方法1, 将jquery及jquery的插件放在前面,然后再引用avalon

<script src='./jquery.js'>
</script>
<script src='./jquery.ui.js'></script>
<script src='./jquery.placehoder.js'></script>
<script src='avalon.js'></script>
<script src='main.js'></script>
<!--main.js为业务代码-->

方法2，将avalon与jquery放在同一目录下，然后：

<!DOCTYPE html>
<html>

<head>
    <title></title>
    <script src="avalon.js" type="text/javascript"></script>
    <script type="text/javascript">
        require.config({
            paths: {
                jquery: "jquery.js",
                avalon: "avalon.js"
            },
            shim: {
                jquery: {
                    exports: "jQuery"
                },
                avalon: {
                    exports: "avalon"
                }
            }
        })
        require(["jquery", "avalon"], function($, avalon) {
            console.log($)
            console.log(avalon)
        })
    </script>
</head>

<body ms-controller="test">
    TEST
</body>

</html>
